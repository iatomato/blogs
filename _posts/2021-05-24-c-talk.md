```c
#include <stdio.h>

struct Node {
        int a;
        int b;
        int c;
};

int main()
{
        /* struct sizeof(s) = 3*sizeof(int) */
        struct Node s = {
                .a = 101,
                .b = 203,
                .c = 301,
        };
        /* int ss[2] = {1,2};         // Ok */
        /* printf("%d\n", *(ss - 1)); //error */
        int *ptr = (int *)(&s + 1);
        int dbg_a = *((int *)(&s + 1) - 1);
        int dbg_b = *((int *)(&s + 1) - 2);
        printf("%p\n%p\t[%p\t%p\t%p]\n",&s, (&s + 1), &(s.a), &(s.b), &(s.c));
        printf("%p\n", (ptr - 1));
        printf("%d\n", *((int *)(&s + 1) - 1));
        return 0;
}
```
